#!/bin/bash

#set job requirements
#SBATCH -N 1 
#SBATCH -p normal
#SBATCH --constraint=gold_6130
#SBATCH -c 16
#SBATCH -t 01:00:00

#download R
module load 2021
module load R/4.1.0-foss-2021a

#run the experiments and save the outcomes for three grahp types, and 2 n's
for graph in "random" "cluster" "scale-free"
    do 
    for n in 50 100
        do
        Rscript --no-save --slave run_and_save.R 50 $n $graph 10 &
        done
    done

#write the outcomes in an output file
for graph in "random" "cluster" "scale-free"
    do 
    for n in 50 100
        do
        Rscript --no-save --slave results_main.R 50 $n $graph 10 &
        done
    done
